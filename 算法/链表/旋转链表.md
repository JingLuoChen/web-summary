# 旋转链表

![mahua](img/旋转链表.jpeg)

````js
var rotateRight = function(head, k) {
    if (k === 0 || !head || !head.next) {
        return head
    }
    let n = 1
    let cur = head
    while (cur) {
        cur = cur.next
        n++
    }
    
    let add = n - k % n
    if (add === n) {
        return head
    }
    
    
    cur.next = head;
    while (add) {
        cur = cur.next;
        add--;
    }

    const ret = cur.next;
    cur.next = null;
    return ret;
       
    
}
````
